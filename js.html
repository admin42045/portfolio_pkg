JavaScript Cheat Sheet

Variables

let name = "John"; const age = 30; var isStudent = true;

Data Types

String, Number, Boolean, Undefined, Null, Object, Array, Symbol (ES6), BigInt

Operators

+ - * / % ** = == === != !== > < >= <= && || ! ++ -- ?: (Ternary)

Functions

function greet(name) { return "Hello " + name; } const greetArrow = (name) => `Hi ${name}`;

Conditionals

if (age > 18) { console.log("Adult"); } else { console.log("Minor"); }

Loops

for (let i = 0; i < 5; i++) { console.log(i); } while (i < 5) { i++; }

Arrays

const fruits = ["apple", "banana"]; fruits.push("orange"); console.log(fruits[1]);

Objects

const person = { name: "John", age: 25 }; console.log(person.name);

DOM Basics

document.getElementById("demo"); document.querySelector(".class"); element.textContent = "Hello";

Events

button.addEventListener("click", function() { alert("Clicked!"); });

ES6 Features

let [a, b] = [1, 2]; const { name, age } = person; const nums = [1, 2, 3]; const newNums = [...nums];

Useful Methods

str.length, str.toUpperCase(), str.includes("x") arr.map(), arr.filter(), arr.reduce() obj.hasOwnProperty("key")

Timers

setTimeout(() => { console.log("Delayed"); }, 1000); setInterval(() => { console.log("Repeats"); }, 1000);

Console

console.log(), console.error(), console.warn(), console.table()

Promises & Async/Await

const p = new Promise((resolve, reject) => { setTimeout(() => resolve("Done"), 1000); }); p.then(result => console.log(result)) .catch(err => console.error(err)); async function asyncFunc() { try { const res = await p; console.log(res); } catch (e) { console.error(e); } }

Classes

class Person { constructor(name, age) { this.name = name; this.age = age; } greet() { return `Hi, I'm ${this.name}`; } } const p = new Person("Alice", 28); console.log(p.greet());

Modules (ES6)

// export.js export const pi = 3.14; export function square(x) { return x * x; } // import.js import { pi, square } from './export.js'; console.log(square(pi));

Template Literals

const name = "John"; const greeting = `Hello, ${name}! How are you today?`; console.log(greeting);

Destructuring with Defaults

const {name = "Guest", age = 30} = {}; console.log(name, age); // Guest 30 const [x = 1, y = 2] = [undefined, 10]; console.log(x, y); // 1 10

Spread & Rest Operators

// Spread in arrays const arr1 = [1, 2]; const arr2 = [...arr1, 3, 4]; // Rest in function args function sum(...nums) { return nums.reduce((a,b) => a+b, 0); } console.log(sum(1,2,3));

Error Handling

try { throw new Error("Oops!"); } catch (e) { console.error(e.message); } finally { console.log("Always runs"); }

Map & Set

const map = new Map(); map.set("a", 1); console.log(map.get("a")); const set = new Set([1,2,2,3]); console.log(set.size); // 3

JSON Methods

const obj = {name: "John", age: 25}; const str = JSON.stringify(obj); const newObj = JSON.parse(str); console.log(newObj.name);

Fetch API (Basic)

fetch('https://api.example.com/data') .then(res => res.json()) .then(data => console.log(data)) .catch(err => console.error(err));
